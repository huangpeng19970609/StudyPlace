### 1. img图片加载失败的回调

```js
<img :src="item.imgUrl" @error="error" />
    
 error(e) {
   let _this = e.target;
   _this.src = "/images/default.gif";
   _this.onerror = null;
 },
```

### 2. 动态加入文件目录的组件

```js
const context = require.context(
  "/public/images/radar/maptime",
  false,
  /\.png$/
);
let imagesArr = [];
context.keys().forEach((item) => {
  imagesArr.push(item.replace(/^\.\//, ""));
});
```
